class Solution {
public:
        /*
1! = 1
2! = 2
3! = 6
4! = 24
5! = 120     <--- 1
...
9! = 362880
10! = 3628800 <--- 2
...
14! = 87178291200
15! = 1307674368000 <--- 3
...
24! = 620448401733239439360000 <-- 4
25! = 15511210043330985984000000 <-- 6 
        */
    int trailingZeroes(int n) {
        if(n < 5) 
            return 0;
        return n / 5 + trailingZeroes(n/5);
    }
};
