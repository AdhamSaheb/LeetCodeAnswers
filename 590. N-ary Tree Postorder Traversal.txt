class Solution {
public:
    vector<int> postorder(Node* root) {
        vector<int> ans; 
        postOrder(root,ans);
        return ans;
    }
    
    void postOrder(Node* root, vector<int> &ans){
        if(!root) return; 
        for(auto child : root->children)
            postOrder(child,ans);
        ans.push_back(root->val);
    }
};
