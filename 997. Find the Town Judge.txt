class Solution {
public:
    int findJudge(int N, vector<vector<int>>& trust) {
        unordered_map<int,bool> mp ;
        unordered_map<int,int> likes ;
        for(vector<int> row : trust) {
            mp[row[0]] =1 ;
            likes[row[1]] ++  ;
        }
        for(int i =1 ;i <N+1 ; i++)
            if(!mp.count(i) && (likes[i] == N-1) ) return i ;
        return -1 ;
    }
};
